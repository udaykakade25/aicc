public class BinarySearch {

    public static Integer binarySearch(List<Integer> arr, Integer key) {
        
        Integer left = 0;
        Integer right = arr.size() - 1;

        while (left <= right) {
            Integer mid = (left + right) / 2;

            if (arr[mid] == key) {
                return mid; 
            }
            else if (arr[mid] < key) {
                left = mid + 1; 
            }
            else {
                right = mid - 1;
            }
        }

        return -1; 
    }

    // Test method
    public static void testSearch() {
        List<Integer> numbers = new List<Integer>{10, 20, 30, 40, 50};
        Integer key = 40;

        Integer result = binarySearch(numbers, key);

        if (result != -1) {
            System.debug('Element found at index: ' + result);
        } else {
            System.debug('Element not found');
        }
    }
}
