public class LinearSearch {

   
    public static Integer linearSearch(List<Integer> arr, Integer key) {
        
        for (Integer i = 0; i < arr.size(); i++) {
            if (arr[i] == key) {
                return i; 
            }
        }
        
        return -1; 
    }

    // Test method
    public static void testSearch() {
        List<Integer> numbers = new List<Integer>{10, 20, 30, 40, 50};
        Integer key = 30;

        Integer result = linearSearch(numbers, key);

        if (result != -1) {
            System.debug('Element found at index: ' + result);
        } else {
            System.debug('Element not found');
        }
    }
}
